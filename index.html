from __future__ import annotations from datetime import datetime, timezone from pathlib import Path from typing import Any import nidaqmx import numpy as np import requests from nidaqmx.constants import ( AcquisitionType, LoggingMode, LoggingOperation, ) from nidaqmx.stream_readers import AnalogMultiChannelReader # ============================================================ # NI SETTINGS # ============================================================ DEVICE_NAME = "cDAQ9181-1C9B71DMod1" CHANNELS = ("ai0", "ai1", "ai2") MIN_VOLTAGE = -10.0 MAX_VOLTAGE = 10.0 RATE_HZ = 400.0 # читаємо стабільно по 1 секунді READ_SECONDS = 1.0 SAMPLES_PER_READ = int(RATE_HZ * READ_SECONDS) # на сайт відправляємо агреговані дані раз на 5 секунд BATCH_SECONDS = 5.0 READS_PER_BATCH = int(BATCH_SECONDS / READ_SECONDS) SAMPLES_PER_BATCH = SAMPLES_PER_READ * READS_PER_BATCH DAQ_BUFFER_SECONDS = 10.0 DAQ_BUFFER_SAMPLES = int(RATE_HZ * DAQ_BUFFER_SECONDS) READ_TIMEOUT_SECONDS = READ_SECONDS + 5.0 # ============================================================ # CLOUDFARE SETTINGS # ============================================================ WORKER_URL = "https://ni-live-monitor.yanovichvitaliy.workers.dev" INGEST_TOKEN = "PUT_YOUR_NEW_ASCII_TOKEN_HERE" # ============================================================ # FILE SETTINGS # ============================================================ BASE_DIR = Path(__file__).resolve().parent DATA_DIR = BASE_DIR / "data" DATA_DIR.mkdir(parents=True, exist_ok=True) # ============================================================ # HELPERS # ============================================================ def utc_now() -> datetime: return datetime.now(timezone.utc) def iso_utc(dt: datetime) -> str: return dt.astimezone(timezone.utc).isoformat(timespec="milliseconds").replace("+00:00", "Z") def make_run_id(dt: datetime) -> str: return dt.strftime("%Y%m%d_%H%M%S") def calc_stats(samples_1d: np.ndarray) -> dict[str, float]: return { "mean": float(np.mean(samples_1d)), "min": float(np.min(samples_1d)), "max": float(np.max(samples_1d)), } def create_tdms_file_path(run_id: str) -> Path: return DATA_DIR / f"ni_measurements_{run_id}.tdms" def build_reset_payload(run_id: str, run_started_at: datetime, tdms_file: Path) -> dict[str, Any]: return { "reset": True, "run_id": run_id, "run_started_at": iso_utc(run_started_at), "updated_at": iso_utc(run_started_at), "file_name": tdms_file.name, "device": DEVICE_NAME, "channels": None, } def build_stats_payload( run_id: str, run_started_at: datetime, tdms_file: Path, batch_data: np.ndarray, batch_end: datetime, ) -> dict[str, Any]: return { "reset": False, "run_id": run_id, "run_started_at": iso_utc(run_started_at), "updated_at": iso_utc(batch_end), "file_name": tdms_file.name, "device": DEVICE_NAME, "rate_hz": RATE_HZ, "window_seconds": BATCH_SECONDS, "channels": { "ai0": calc_stats(batch_data[0]), "ai1": calc_stats(batch_data[1]), "ai2": calc_stats(batch_data[2]), }, } def send_payload(session: requests.Session, payload: dict[str, Any]) -> None: response = session.post( WORKER_URL + "/ingest", headers={ "Authorization": f"Bearer {INGEST_TOKEN}", "Content-Type": "application/json", }, json=payload, timeout=10, ) response.raise_for_status() # ============================================================ # MAIN # ============================================================ def main() -> None: if INGEST_TOKEN == "PUT_YOUR_NEW_ASCII_TOKEN_HERE": raise RuntimeError("Replace INGEST_TOKEN with your real ASCII token.") run_started_at = utc_now() run_id = make_run_id(run_started_at) tdms_file = create_tdms_file_path(run_id) print(f"Run ID: {run_id}") print(f"TDMS file: {tdms_file}") session = requests.Session() # одразу повідомляємо сервер, що почався новий запуск try: reset_payload = build_reset_payload(run_id, run_started_at, tdms_file) send_payload(session, reset_payload) print("Reset payload sent.") except Exception as e: print("Reset send error:", e) read_buffer = np.empty((len(CHANNELS), SAMPLES_PER_READ), dtype=np.float64) batch_buffer = np.empty((len(CHANNELS), SAMPLES_PER_BATCH), dtype=np.float64) batch_fill = 0 with nidaqmx.Task() as task: for channel in CHANNELS: task.ai_channels.add_ai_voltage_chan( f"{DEVICE_NAME}/{channel}", min_val=MIN_VOLTAGE, max_val=MAX_VOLTAGE, ) task.timing.cfg_samp_clk_timing( rate=RATE_HZ, sample_mode=AcquisitionType.CONTINUOUS, samps_per_chan=DAQ_BUFFER_SAMPLES, ) # TDMS logging: NI-DAQmx пише сирі дані прямо у файл task.in_stream.configure_logging( tdms_file, logging_mode=LoggingMode.LOG_AND_READ, group_name="NI", operation=LoggingOperation.OPEN_OR_CREATE, ) reader = AnalogMultiChannelReader(task.in_stream) task.start() print("Measurement started.") while True: # 1 секунда сирих даних = 400 точок/канал reader.read_many_sample( read_buffer, number_of_samples_per_channel=SAMPLES_PER_READ, timeout=READ_TIMEOUT_SECONDS, ) chunk_data = read_buffer.copy() # додаємо у 5-секундний буфер start_index = batch_fill end_index = batch_fill + SAMPLES_PER_READ batch_buffer[:, start_index:end_index] = chunk_data batch_fill = end_index # якщо ще не накопичили 5 секунд — чекаємо далі if batch_fill < SAMPLES_PER_BATCH: continue # маємо 5 секунд даних — рахуємо mean/min/max batch_end = utc_now() batch_data = batch_buffer.copy() batch_fill = 0 stats_payload = build_stats_payload( run_id=run_id, run_started_at=run_started_at, tdms_file=tdms_file, batch_data=batch_data, batch_end=batch_end, ) try: send_payload(session, stats_payload) print(f"Sent stats for run {run_id} at {stats_payload['updated_at']}") except Exception as e: print("Send error:", e) if __name__ == "__main__": main()
